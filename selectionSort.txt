#include<stdio.h>

void main()
{
          int n,x, a[100],pos;
          ///////////////////////////   INPUT   /////////////////////////
          printf("enter the number of elements :");
          scanf("%d",&n);

          for(int i=0;i<n;i++)
          scanf("%d",&a[i]);
          
          for(int i=0;i<n-1;i++)
          {
                pos=i;
                for(int j=i+1;j<n;j++)
                {
                      if(a[pos]>a[j])
                       pos=j;
                }
                if(pos!=i)
                {
                      int t=a[i];
                      a[i]=a[pos];

          }
