// C program to traverse the array
#include <stdio.h>
void main()
{
	int a[100],i,n;
	printf("enter the number of elements :");
	scanf("%d",&n);
	printf("enter the elements :");
	for(i=0;i<n;i++)
	{
	scanf("%d",&a[i]);
	}
	printf("array elements are :");
	i=0;
	for(i=0;i<n;i++)
	{
	    printf(" %d",a[i]);
	}
	printf(" \n reversed array :");
	for(i=n-1;i>=0;i--)
	{
	    printf("%d",a[i]);
	    printf(" ");
	}
}
