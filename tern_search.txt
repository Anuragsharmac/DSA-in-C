     int ternarySearch(int l,int h,int x,int a[])
     {
          int mid1,mid2,k=0;
            if(l<=h)
           {
              mid1=l+((l-h)/3);
              mid2=h+((h-l)/3);
              if(a[mid1]==x)
              {
                   k=mid1;
                  return 1;
               }
              else if( a[mid2]==x)
              {
                   k=mid2;
               return 1;
              }
              
              if ( x < a[mid1])
              {
                h=mid1-1;
                return ternarySearch(l, mid1 - 1, x, a);
               }
               else if (x > a[mid2]) 
               {
                l=mid2+1;
                return ternarySearch(mid2 + 1, h, x, a);
                }
                else if(x<a[mid2] && x>a[mid1])
                {
                l=mid1+1;
                h=mid2-1;
                return ternarySearch(mid1 + 1, mid2 - 1, x, a);
                }
                else
                return 0;
           }
           if(1)
          {
               printf("found ");
          }
          else
          printf("not found");
          return 0;
     }
