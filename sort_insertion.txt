#include <stdio.h>
void main()
{
    int a[100],i,c=0,m,n;
    //input taking
    printf("\n enter the number of elements :");
    scanf("%d",&n);
    printf("\n enter the elements of array :");
    for(i=0;i<n;i++)
    { 
        scanf("%d",&a[i]);
    }
    //element to be added
    printf("\n enter the element to be added in the sorted array :");
    scanf("%d",&m);
    printf("\n original array :");
    //searching the position of the element to be added
    for(int j=0;j <=n;j++)
    {
        printf("%d",a[j]);
        printf(" ");
        if(a[j]<m)
        {
            c++;
        }
        else
        {
            break;
        }
    }
    printf("\n pos: %d",c+1);
       for(i=n-1;i>=c;i--)
      {
        a[i+1]=a[i];
      }
      a[c]=m;
      printf("\n new array: ");
      for(i=0;i<=n;i++)
      {
          printf("%d",a[i]); 
          printf(" ");
      }
}
